local Players = game:GetService("Players")

local player = Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local hrp = character:WaitForChild("HumanoidRootPart")

local toggled = false
local bodyGyro = nil
local bodyVelocity = nil
local teleportCoroutine = nil
local moveSpeed = 65 -- movement speed (can tweak)

local function equipSpeedCoil()
    local backpack = player:WaitForChild("Backpack")
    local char = player.Character
    if char then
        local tool = char:FindFirstChild("Speed Coil")
        if tool then
            tool.Parent = backpack -- unequip first
        end
    end
    local coilTool = backpack:FindFirstChild("Speed Coil")
    if coilTool then
        coilTool.Parent = char -- equip
    end
end

local function setupBodyControls()
    if bodyGyro then bodyGyro:Destroy() end
    if bodyVelocity then bodyVelocity:Destroy() end

    bodyGyro = Instance.new("BodyGyro")
    bodyGyro.MaxTorque = Vector3.new(400000, 400000, 400000)
    bodyGyro.P = 10000
    bodyGyro.D = 1000
    bodyGyro.CFrame = hrp.CFrame
    bodyGyro.Parent = hrp

    bodyVelocity = Instance.new("BodyVelocity")
    bodyVelocity.MaxForce = Vector3.new(400000, 400000, 400000)
    bodyVelocity.Velocity = Vector3.new(0,0,0)
    bodyVelocity.Parent = hrp
end

local function clearBodyControls()
    if bodyGyro then bodyGyro:Destroy() bodyGyro = nil end
    if bodyVelocity then bodyVelocity:Destroy() bodyVelocity = nil end
end

local function lookAt(targetPos)
    local direction = (targetPos - hrp.Position)
    if direction.Magnitude == 0 then return end
    local unitDirection = direction.Unit
    local targetCFrame = CFrame.new(hrp.Position, hrp.Position + unitDirection)
    if bodyGyro then
        bodyGyro.CFrame = targetCFrame
    end
end

local function moveToPosition(targetPos, speed)
    speed = speed or moveSpeed
    while toggled do
        local direction = targetPos - hrp.Position
        local dist = direction.Magnitude
        if dist < 1 then
            break
        end

        lookAt(targetPos)
        if bodyVelocity then
            bodyVelocity.Velocity = direction.Unit * speed
        end
        task.wait(0) -- snappier update
    end
    if bodyVelocity then
        bodyVelocity.Velocity = Vector3.new(0,0,0) -- stop instantly
    end
end

local function getNearestPlot()
    local plots = workspace:FindFirstChild("Plots")
    if not plots then return nil end

    local nearestPlot = nil
    local nearestDistance = math.huge
    for _, plot in pairs(plots:GetChildren()) do
        if plot:IsA("Model") and plot:FindFirstChild("AnimalPodiums") then
            local plotCFrame = nil
            if plot.PrimaryPart then
                plotCFrame = plot.PrimaryPart.CFrame
            else
                local parts = plot:GetDescendants()
                local positions = {}
                for _, p in ipairs(parts) do
                    if p:IsA("BasePart") then
                        table.insert(positions, p.Position)
                    end
                end
                if #positions > 0 then
                    local sum = Vector3.new(0,0,0)
                    for _, pos in ipairs(positions) do sum += pos end
                    plotCFrame = CFrame.new(sum / #positions)
                end
            end

            if plotCFrame then
                local dist = (hrp.Position - plotCFrame.Position).Magnitude
                if dist < nearestDistance then
                    nearestDistance = dist
                    nearestPlot = plot
                end
            end
        end
    end
    return nearestPlot
end

local function getPodiumModel(index)
    local plot = getNearestPlot()
    if not plot then return nil end
    local podiums = plot:FindFirstChild("AnimalPodiums")
    if not podiums then return nil end
    local podiumModel = podiums:FindFirstChild(tostring(index))
    return podiumModel
end

local function getPodiumCFrame(podiumModel)
    if not podiumModel then return nil end
    if podiumModel.PrimaryPart then
        return podiumModel.PrimaryPart.CFrame
    else
        local parts = podiumModel:GetDescendants()
        local sum = Vector3.new(0,0,0)
        local count = 0
        for _, p in ipairs(parts) do
            if p:IsA("BasePart") then
                sum += p.Position
                count += 1
            end
        end
        if count > 0 then
            local avgPos = sum / count
            return CFrame.new(avgPos)
        else
            return nil
        end
    end
end

local function teleportLoop()
    local sequence = {
        1,2,3,4,5,
        6,5,6,
        7,8,9,10,
        10,6,5,
        4,3,2,1
    }

    while toggled do
        for _, idx in ipairs(sequence) do
            if not toggled then break end
            local podiumModel = getPodiumModel(idx)
            local targetCFrame = getPodiumCFrame(podiumModel)
            if targetCFrame then
                local targetPos = targetCFrame.Position + Vector3.new(0, 5, 0) -- 3 studs above actual pos
                moveToPosition(targetPos, moveSpeed)
            else
                warn("Podium model "..tostring(idx).." missing or no CFrame")
            end
            task.wait(0)
        end
    end
    clearBodyControls()
end

local function startTeleportLoop()
    setupBodyControls()
    if teleportCoroutine then
        pcall(function() coroutine.close(teleportCoroutine) end)
    end
    teleportCoroutine = coroutine.create(teleportLoop)
    coroutine.resume(teleportCoroutine)
end

local function onCharacterAdded(char)
    character = char
    hrp = character:WaitForChild("HumanoidRootPart")

    equipSpeedCoil()

    if toggled then
        setupBodyControls()
        if teleportCoroutine then
            pcall(function() coroutine.close(teleportCoroutine) end)
        end
        teleportCoroutine = coroutine.create(teleportLoop)
        coroutine.resume(teleportCoroutine)
    end
end

player.CharacterAdded:Connect(onCharacterAdded)

equipSpeedCoil()

Main:AddToggle({
    Name = "anti hit [base]",
    Default = false,
    Callback = function(value)
        toggled = value
        if toggled then
            if not character or not hrp then
                character = player.Character or player.CharacterAdded:Wait()
                hrp = character:WaitForChild("HumanoidRootPart")
            end

            equipSpeedCoil()
            startTeleportLoop()
        else
            clearBodyControls()
            if teleportCoroutine then
                pcall(function() coroutine.close(teleportCoroutine) end)
                teleportCoroutine = nil
            end
        end
    end
})
local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer
local RunService = game:GetService("RunService")

local antiHitEnabled = false
local loopConnection
local bodyGyro
local bodyVelocity

Main:AddToggle({
	Name = "anti hit [outside]",
	Default = false,
	Callback = function(Value)
		antiHitEnabled = Value

		-- Cleanup if disabling
		if not Value then
			if loopConnection then loopConnection:Disconnect() loopConnection = nil end
			if bodyGyro then bodyGyro:Destroy() bodyGyro = nil end
			if bodyVelocity then bodyVelocity:Destroy() bodyVelocity = nil end
			return
		end

		-- Main loop
		loopConnection = RunService.Heartbeat:Connect(function()
			if not antiHitEnabled then return end

			local char = LocalPlayer.Character
			if not char or not char:FindFirstChild("HumanoidRootPart") then return end
			local root = char.HumanoidRootPart

			local nearest
			local nearestDist = math.huge

			-- Find nearest player
			for _, plr in ipairs(Players:GetPlayers()) do
				if plr ~= LocalPlayer and plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
					local dist = (plr.Character.HumanoidRootPart.Position - root.Position).Magnitude
					if dist < nearestDist then
						nearestDist = dist
						nearest = plr
					end
				end
			end

			if nearest and nearestDist <= 25 then
				-- Create movement parts if not exist
				if not bodyGyro then
					bodyGyro = Instance.new("BodyGyro")
					bodyGyro.MaxTorque = Vector3.new(1e5, 1e5, 1e5)
					bodyGyro.P = 1e5
					bodyGyro.Parent = root
				end
				if not bodyVelocity then
					bodyVelocity = Instance.new("BodyVelocity")
					bodyVelocity.MaxForce = Vector3.new(1e5, 1e5, 1e5)
					bodyVelocity.P = 1e4
					bodyVelocity.Parent = root
				end

				-- Fly away from nearest
				local awayDir = (root.Position - nearest.Character.HumanoidRootPart.Position).Unit
				bodyGyro.CFrame = CFrame.lookAt(root.Position, root.Position + awayDir)
				bodyVelocity.Velocity = awayDir * 65
			else
				-- Stop movement but keep toggle ON
				if bodyGyro then bodyGyro:Destroy() bodyGyro = nil end
				if bodyVelocity then bodyVelocity:Destroy() bodyVelocity = nil end
			end
		end)
	end
})
local UserInputService = game:GetService("UserInputService")
local Players = game:GetService("Players")

local player = Players.LocalPlayer

local humanoid = character:WaitForChild("Humanoid")
local rootPart = character:WaitForChild("HumanoidRootPart")

local platformSpawnEnabled = false

-- Update refs on respawn
player.CharacterAdded:Connect(function(char)
    character = char
    humanoid = character:WaitForChild("Humanoid")
    rootPart = character:WaitForChild("HumanoidRootPart")
end)

local function createPlatformUnder()
    if not rootPart then return end

    local platform = Instance.new("Part")
    platform.Size = Vector3.new(1, 1, 1)
    platform.Anchored = true
    platform.CanCollide = true
    platform.Transparency = 0.8
    platform.Material = Enum.Material.Neon
    platform.BrickColor = BrickColor.new("Really black")

    platform.CFrame = rootPart.CFrame * CFrame.new(0, -3, 0)
    platform.Parent = workspace

    Debris:AddItem(platform, 1) -- now deletes after 1 second
end

Other:AddToggle({

    Callback = function(value)
        platformSpawnEnabled = value
        print("pp", value)
    end
})

-- Listen for jump only when toggle is ON
UserInputService.JumpRequest:Connect(function()
    if platformSpawnEnabled and humanoid and rootPart then
        createPlatformUnder()
    end
end)


-- Store these outside so they persist between toggle changes
local bodyGyro, bodyVelocity, flyConnection

Main:AddToggle({
	Name = "boost for steal",
	Default = false,
	Callback = function(Value)
		local Players = game:GetService("Players")
		local RunService = game:GetService("RunService")
		local UIS = game:GetService("UserInputService")
		local LocalPlayer = Players.LocalPlayer

		local flySpeed = 40

		if Value then
			local Character = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
			local Root = Character:FindFirstChild("HumanoidRootPart")
			local Humanoid = Character:FindFirstChildWhichIsA("Humanoid")

			if Root and Humanoid then
				Humanoid.PlatformStand = true

				-- Create BodyGyro
				bodyGyro = Instance.new("BodyGyro")
				bodyGyro.MaxTorque = Vector3.new(1e5, 1e5, 1e5)
				bodyGyro.P = 1e4
				bodyGyro.CFrame = Root.CFrame
				bodyGyro.Parent = Root

				-- Create BodyVelocity
				bodyVelocity = Instance.new("BodyVelocity")
				bodyVelocity.MaxForce = Vector3.new(1e5, 1e5, 1e5)
				bodyVelocity.Velocity = Vector3.zero
				bodyVelocity.Parent = Root

				-- Fly movement loop
				flyConnection = RunService.RenderStepped:Connect(function()
					local camCF = workspace.CurrentCamera.CFrame
					bodyGyro.CFrame = camCF

					local moveDir = Humanoid.MoveDirection

					-- Up/Down (PC keys unless mobile GUI is added)
					if UIS:IsKeyDown(Enum.KeyCode.Space) then
						moveDir += Vector3.new(0, 1, 0)
					end
					if UIS:IsKeyDown(Enum.KeyCode.LeftControl) then
						moveDir -= Vector3.new(0, 1, 0)
					end

					bodyVelocity.Velocity = moveDir * flySpeed
				end)
			end
		else
			-- Disable fly
			local Character = LocalPlayer.Character
			if Character then
				local Humanoid = Character:FindFirstChildWhichIsA("Humanoid")
				if Humanoid then
					Humanoid.PlatformStand = false
				end
			end

			if flyConnection then
				flyConnection:Disconnect()
				flyConnection = nil
			end
			if bodyGyro then
				bodyGyro:Destroy()
				bodyGyro = nil
			end
			if bodyVelocity then
				bodyVelocity:Destroy()
				bodyVelocity = nil
			end
		end
	end    
})
